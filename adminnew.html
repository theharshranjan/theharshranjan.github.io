PWA — Install as AppRecommended
📱 On Android Chrome / Desktop Chrome & Edge, an "Install" button will appear above (and in the top toolbar) once the app is installable — one tap and it's on your home screen / app list, no browser bar.

iOS Safari doesn't support the auto-install prompt: tap Share → "Add to Home Screen" instead.

For the main website (thesamarpan.co.in) the PWA install button appears automatically after scrolling — or via browser menu.

Three files must sit in the same folder as admin.html: manifest-admin.json, sw-admin.js (service worker — required for the install prompt to appear), and the icon files (favicon-192x192.png, favicon-512x512.png, apple-touch-icon.png). Google Sheets Sync — Visitor LogOff
📄 Every visitor session already saved to Firestore also gets pushed into a Google Sheet, batched every 60 seconds — so you always have a plain-spreadsheet copy of the data with zero storage worries on Firestore's side. Requires a small (free) Google Apps Script Web App — steps below.
  1. Create a new Google Sheet (any name).
  2. In it: Extensions → Apps Script, delete the sample code, paste the script from the "Apps Script Code" box below.
  3. Deploy → New deployment → Web app. Execute as: Me. Who has access: Anyone.
  4. Copy the Web App URL it gives you and paste it below.
Not configured yet.
Apps Script Code (click to expand, then copy) 
function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('Visitors') || ss.insertSheet('Visitors');
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['ID','Date','Time','Page','Device','IP Address','City','Region','Country','Referrer','Duration (sec)']);
  }
  var data = JSON.parse(e.postData.contents);
  (data.rows || []).forEach(function(r){ sheet.appendRow(r); });
  return ContentService.createTextOutput(JSON.stringify({status:'ok', added: (data.rows||[]).length}))
    .setMimeType(ContentService.MimeType.JSON);
}

Firestore Rules (Updated — all collections)Updated

Paste in Firebase → Firestore → Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /homepage_leads/{docId} {
      allow create: if isValidLead();
      allow read, update, delete: if isAdmin();
    }
    match /homepage_newsletter/{docId} {
      allow create: if isValidEmail();
      allow read, update, delete: if isAdmin();
    }
    match /homepage_quiz_results/{docId} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }
    match /homepage_visitors/{docId} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }
    match /homepage_yoga_sessions/{docId} {
      allow create: if isValidEmail();
      allow read, update, delete: if isAdmin();
    }
    match /homepage_referrals/{docId} {
      allow create: if isValidEmail();
      allow read, update, delete: if isAdmin();
    }
    match /homepage_bookings/{docId} {
      allow create: if isValidLead();
      allow read, update, delete: if isAdmin();
    }
    match /homepage_about_leads/{docId} {
      allow create: if isValidLead();
      allow read, update, delete: if isAdmin();
    }
    match /samarpan_journal_posts/{docId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }
    match /samarpan_flash_cards/{docId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }
    match /samarpan_faqs/{docId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }
    match /samarpan_testimonials/{docId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }
    match /samarpan_site_content/{docId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }
    match /samarpan_videos/{docId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }
    match /samarpan_community_posts/{docId} {
      allow read: if true;
      allow create: if true;
      allow update, delete: if isAdmin();
    }
    match /samarpan_admin_notes/{docId} {
      allow read, write: if isAdmin();
    }
    match /samarpan_admin_users/{docId} {
      allow read, write: if isAdmin();
    }
    match /samarpan_login_history/{docId} {
      allow read, write: if isAdmin();
    }
    match /{document=**} {
      allow read, write: if false;
    }
    function isAdmin() {
      return request.auth != null;
    }
    function isValidEmail() {
      return request.resource.data.keys().hasAny(['email'])
        && request.resource.data.email is string
        && request.resource.data.email.size() > 4
        && request.resource.data.email.size() < 200;
    }
    function isValidLead() {
      return request.resource.data.keys().hasAll(['email','name'])
        && request.resource.data.email is string
        && request.resource.data.name is string
        && request.resource.data.email.size() > 4
        && request.resource.data.name.size() > 0;
    }
  }
}
Storage Rules (Owner Photo Upload)New

Paste in Firebase → Storage → Rules so the photo-upload button above works.

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /owner-photos/{fileName} {
      allow read: if true;
      allow write: if request.auth != null
        && request.resource.size < 3 * 1024 * 1024
        && request.resource.contentType.matches('image/.*');
    }
    match /media/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
